## Queue를 이용한 방법.
6848KB,	264ms
```C#
class Program
{
    private const int bufferSize = 131072; //131072=128KB. 경험적 가장빠른속도. 기본=4096=4KB.
    public static readonly StreamReader sr = new(new BufferedStream(Console.OpenStandardInput(), bufferSize));
    public static readonly StreamWriter sw = new(new BufferedStream(Console.OpenStandardOutput(), bufferSize));

    static void Main()
    {
        string[] s = sr.ReadLine().Split();
        int n = int.Parse(s[0]);
        int k = int.Parse(s[1]);

        Queue<int> queue = new Queue<int>();
        for (int i = 1; i <= n; i++)
        {
            queue.Enqueue(i);
        }

        sw.Write("<");
        while (queue.Count > 1)
        {
            for (int i = 0; i < k - 1; i++)
            {
                queue.Enqueue(queue.Dequeue());
            }

            sw.Write($"{queue.Dequeue()}, ");
        }

        sw.Write($"{queue.Dequeue()}>");

        sw.Flush();
    }
}
```
## LinkedList를 이용한 방법.
6792KB,	136ms
```C#
class Program
{
    private const int bufferSize = 131072; //131072=128KB. 경험적 가장빠른속도. 기본=4096=4KB.
    public static readonly StreamReader sr = new(new BufferedStream(Console.OpenStandardInput(), bufferSize));
    public static readonly StreamWriter sw = new(new BufferedStream(Console.OpenStandardOutput(), bufferSize));

    static void Main()
    {
        string[] s = sr.ReadLine().Split();
        int n = int.Parse(s[0]);
        int k = int.Parse(s[1]);

        var list = new LinkedList<int>();
        for (int i = 1; i <= n; i++)
        {
            list.AddLast(i);
        }

        var cur = list.First;
        
        sw.Write("<");
        while (list.Count > 1)
        {
            for (int i = 0; i < k - 1; i++)
            {
                cur = cur.Next ?? list.First;
            }
            sw.Write($"{cur.Value}, ");
            var next = cur.Next ?? list.First;
            list.Remove(cur);
            cur = next;
        }
        
        sw.Write($"{cur.Value}>");

        sw.Flush();
    }
}
```
알게된점.
`list.Remove(cur);`를 실행하면, cur.Previous와 cur.Next는 null이 된다.
cur.Value는 살아있다.



## List<T>를 이용한 방법.
6064KB,	56ms
```C#
class Program
{
    private const int bufferSize = 131072; //131072=128KB. 경험적 가장빠른속도. 기본=4096=4KB.
    public static readonly StreamReader sr = new(new BufferedStream(Console.OpenStandardInput(), bufferSize));
    public static readonly StreamWriter sw = new(new BufferedStream(Console.OpenStandardOutput(), bufferSize));

    static void Main()
    {
        string[] s = sr.ReadLine().Split();
        int n = int.Parse(s[0]);
        int k = int.Parse(s[1]);

        List<int> list = new List<int>();
        sw.Write("<");

        for (int i = 1; i <= n; i++)
        {
            list.Add(i);
        }

        int index = 0;

        while (list.Count > 0)
        {
            index = (index + k - 1) % list.Count;

            sw.Write(list[index]);
            list.RemoveAt(index);

            if (list.Count > 0)
            {
                sw.Write(", ");
            }
        }

        sw.Write(">");
        sw.Flush();
    }
}
```
## array를 이용한 방법.
6052KB,	56ms
```C#
class Program
{
    private const int bufferSize = 131072; //131072=128KB. 경험적 가장빠른속도. 기본=4096=4KB.
    public static readonly StreamReader sr = new(new BufferedStream(Console.OpenStandardInput(), bufferSize));
    public static readonly StreamWriter sw = new(new BufferedStream(Console.OpenStandardOutput(), bufferSize));

    static void Main()
    {
        string[] s = sr.ReadLine().Split();
        int n = int.Parse(s[0]);
        int k = int.Parse(s[1]);

        int[] arr = new int[n];
        for (int i = 0; i < n; i++)
        {
            arr[i] = i + 1;
        }

        sw.Write("<");
        int index = 0;
        int size = n; //배열크기.
        while (size > 0)
        {
            index = (index + k - 1) % size;

            sw.Write(arr[index]);

            size--; //배열크기감소

            if (size > 0)
            {
                sw.Write(", ");

                //index위치의 제거로, 하나씩 당기기.
                for (int i = index; i < size; i++)
                {
                    arr[i] = arr[i + 1];
                }
            }
        }


        sw.Write(">");
        sw.Flush();
    }
}
```
